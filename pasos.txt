## Crear tablas

CREATE TABLE productos (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100),
    precio INT,
    descripcion TEXT
);

CREATE TABLE pedidos (
    id SERIAL PRIMARY KEY,
    cliente VARCHAR(100),
    total INT,
    fecha DATETIME
);

CREATE TABLE pedido_productos (
    pedido_id INT,
    producto_id INT,
    cantidad INT
);
